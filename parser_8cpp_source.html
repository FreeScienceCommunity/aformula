00177 {

00179     while (isspace (parseBuffer[0]))
00180         parseBuffer.erase (0, 1);
00181 
00182     // Bail if we're done
00183     if (!parseBuffer.length ())
00184         return TOKEN_END;
00185 
00186     // See if the first part of the parseBuffer is an operator string
00187     for (int i = 0 ; i < NUM_OPERATORS ; i++)
00188     {
00189         if (parseBuffer.compare (0, strlen (operators[i].name), operators[i].name) == 0)
00190         {
00191             // Indeed, an operator is here
00192             strToken.erase ();
00193 
00194             strToken = operators[i].name;
00195             parseBuffer.erase (0, strlen (operators[i].name));
00196             
00197             return TOKEN_OPERATOR;
00198         }
00199     }
00200     
00201     // If the next thing begins with a letter, it's some kind of
00202     // identifier
00203     if (isalpha (parseBuffer[0]))
00204     {
00205         strToken.erase ();
00206             
00207         while (parseBuffer.length () &&
00208                (isalpha (parseBuffer[0]) || isdigit (parseBuffer[0])))
00209         {
00210             strToken += parseBuffer[0];
00211             parseBuffer.erase (0,1);
00212         }
00213             
00214         return getIdentifierType (strToken);
00215     }
00216         
00217     // Neither operator nor identifier, try number
00218     bool haveDot = false;
00219     if (isdigit (parseBuffer[0]) || parseBuffer[0] == '.')
00220     {
00221         strToken.erase ();
00222             
00223         while (parseBuffer.length () &&
00224                (isdigit (parseBuffer[0]) || parseBuffer[0] == '.'))
00225         {
00226             strToken += parseBuffer[0];
00227                 
00228             // Two decimal points?
00229             if (parseBuffer[0] == '.' && haveDot)
00230                 return TOKEN_BAD;
00231             if (parseBuffer[0] == '.')
00232                 haveDot = true;
00233                 
00234             parseBuffer.erase (0, 1);
00235         }
00236             
00237         // Convert the string token to a decimal
00238         const char *str = strToken.c_str ();

00240         numToken = strtod (str, &endptr);
00241             
00242         // Check for failure
00243         if (numToken == 0.0 && endptr == str)
00244             return TOKEN_BAD;
00245             
00246         return TOKEN_NUMBER;
00247     }
00248         
00249     // Syntactic glue that doesn't parse into the tree
00250     if (parseBuffer[0] == '(')
00251     {
00252         strToken = parseBuffer[0];
00253         parseBuffer.erase (0,1);
00254         return TOKEN_PAREN_OPEN;
00255     }
00256         
00257     if (parseBuffer[0] == ')')
00258     {
00259         strToken = parseBuffer[0];
00260         parseBuffer.erase (0,1);
00261         return TOKEN_PAREN_CLOSE;
00262     }
00263         
00264     if (parseBuffer[0] == ',')
00265     {
00266         strToken = parseBuffer[0];

00269     }

00473 ExprAST *Parser::parsePrimary ()
00474 {
00475     // At the top level, we have to have either an identifier of some kind,
00476     // a number, or an open paren.  We might also have a unary minus sign, handle
00477     // that specially right here.
00478     switch (currentToken)
00479     {
00480     case TOKEN_IDENTIFIER_FUNCTION:
00481         return parseFunctionIdentifierExpr ();
00482     case TOKEN_IDENTIFIER_CONSTANT:
00483         return parseConstantIdentifierExpr ();
00484     case TOKEN_IDENTIFIER_VARIABLE:
00485         return parseVariableIdentifierExpr ();
00486     case TOKEN_NUMBER:
00487         return parseNumberExpr ();
00488     case TOKEN_PAREN_OPEN:
00489         return parseOpenParenExpr ();
00490     case TOKEN_OPERATOR:

00493         // Fall through for any other operator

00495         error (boost::format ("Expecting an expression, got %1% instead!") %
00496                formatToken ());

00501 ExprAST *Parser::parseBinOpRHS (int exprPrecedence, ExprAST *LHS)
00502 {
00503     while (1)
00504     {
00505         // This should be either an operator, a close-paren, comma,
00506         // or the end of the formula.
00507         if (currentToken != TOKEN_OPERATOR &&
00508             currentToken != TOKEN_PAREN_CLOSE &&
00509             currentToken != TOKEN_COMMA &&
00510             currentToken != TOKEN_END)
00511         {
00512             error (boost::format ("Attempting to parse the RHS of a binary operator, "
00513                                   "and got %1%!") % formatToken ());
00514             return NULL;
00515         }
00516             
00517         // Get the precedence
00518         int tokenPrecedence = -1;
00519             
00520         for (int i = 0 ; i < NUM_OPERATORS ; i++)
00521         {
00522             if (strToken == operators[i].name)
00523             {
00524                 tokenPrecedence = operators[i].precedence;
00525                 break;
00526             }
00527         }
00528             
00529         // We're done if we run into an equivalent or lower-precedence binary
00530         // operator (or, notably, something that's not an operator at all, like
00531         // the end of the formula)
00532         if (tokenPrecedence < exprPrecedence)
00533             return LHS;
00534             
00535         // Pull off the operator
00536         std::string binOp = strToken;
00537         getNextToken ();
00538             
00539         // Parse the primary after the operator
00540         ExprAST *RHS = parsePrimary ();
00541         if (!RHS)
00542             return NULL;
00543             
00544         // We've now parsed a full binary operator pair, "A op B".  If the next
00545         // thing is a binary operator (which it should be, or it should be
00546         // nothing at all), we need to see whether we want "(A op B) op2 ..." or
00547         // "A op (B op2 ...)" by precedence rules.  Look ahead:
00548         int nextPrecedence = -1;
00549         for (int i = 0 ; i < NUM_OPERATORS ; i++)
00550         {
00551             if (strToken == operators[i].name)
00552             {
00553                 nextPrecedence = operators[i].precedence;
00554                 break;
00555             }
00556         }
00557             
00558         if (tokenPrecedence < nextPrecedence)
00559         {
00560             // Next precedence is greater, we want "A op (B op2 ...)".  Do that by
00561             // parsing B op2 ... into a new RHS.
00562             RHS = parseBinOpRHS (tokenPrecedence + 1, RHS);
00563             if (RHS == NULL)
00564                 return NULL;
00565         }
00566             
00567         // Next precedence is lower, we want "(A op B) op2 ...", so make a new LHS
00568         LHS = new BinaryExprAST (binOp, LHS, RHS);
00569             
00570         // And loop back to the beginning
00571     }
00572 }
