00147 int Formula::fastestBackend (bool setAsDefault)
00148 {

00150     double bestBackendTime = 9999999.0;

00152     // Trap SIGABRT, SIGFPE, SIGILL, SIGSEGV, and don't let a rogue
00153     // backend bring down the entire system
00154     signal (SIGABRT, signal_handler);
00155     signal (SIGFPE, signal_handler);
00156     signal (SIGILL, signal_handler);
00157     signal (SIGSEGV, signal_handler);
00158 
00159     for (int backend = 1 ; backend < NUM_BACKENDS ; backend++)
00160     {
00161         // Reset the signal error flag
00162         errorflag = 0;
00163         
00164         double timeStart = timerTime ();
00165         
00166         Formula *formula = Formula::createFormula (backend);
00167         if (!formula || errorflag)
00168             continue;
00169 
00170         double x;
00171         formula->setVariable ("x", &x);
00172         if (errorflag)
00173             continue;
00174 
00175         formula->setExpression ("x*x + 4");
00176         if (errorflag)
00177             continue;
00178 
00179         for (int i = 0 ; i < 1000 ; i++)
00180         {
00181             x = i;
00182             double ret = formula->evaluate ();
00183             if (errorflag)
00184                 break;
00185             if (ret != i * i + 4)
00186             {
00187                 errorflag = 1;
00188                 break;
00189             }
00190         }
00191 
00192         if (errorflag)
00193             continue;
00194         
00195         delete formula;
00196         
00197         double timeEnd = timerTime ();
00198 
00199         if (timeEnd - timeStart < bestBackendTime)
00200         {
00201             bestBackendTime = timeEnd - timeStart;
00202             bestBackend = backend;
00203         }
00204     }
00205 
00206     // Release the signals
00207     signal (SIGABRT, SIG_DFL);
00208     signal (SIGFPE, SIG_DFL);
00209     signal (SIGILL, SIG_DFL);
00210     signal (SIGSEGV, SIG_DFL);
